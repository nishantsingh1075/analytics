# Enter your code here. Read input from STDIN. Print output to STDOUT

input_str = str(input())
in_list = input_str.split()
n = int(in_list[0])
m = n*3

hr = n//2
hc = m//2

for i in range(hr):
    print(".|".rjust(hc+1-(3*i),"-")+"..|"*(i*2)+".".ljust(hc-(3*i),"-"))

print("WELCOME".center(m, "-"))

for i in range(hr):
    print(".|".rjust(hc-(3*(hc-i-1)),"-")+"..|"*(3*(hc-i-1))+".".ljust(hc-(3*(hc-i-1)),"-"))
